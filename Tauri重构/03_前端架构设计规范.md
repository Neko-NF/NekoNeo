# NekoNeo — 前端架构设计规范（Vue 3 + Vite + TypeScript）

> 版本：v1.0 | 日期：2026-05-10

---

## 一、总体原则

1. **单向数据流**：视图读 Store → 操作调 Composable → Composable 调 `invoke()` → 结果更新 Store
2. **Views 不含业务逻辑**：页面组件只负责布局和事件绑定，业务逻辑下沉到 Composable 或 Store
3. **类型完整**：所有 Tauri Command 的参数和返回值必须有对应 TypeScript 类型定义
4. **组件单一职责**：每个组件只做一件事，超过 200 行的 SFC 需考虑拆分
5. **禁止内联样式**：颜色、间距、圆角全部使用 CSS 变量，不允许 `:style="{ color: '#xxx' }"`

## 四、Pinia Store 规范

### 4.1 Store 模板

```typescript
// src/stores/service.ts

import { defineStore } from 'pinia'
import { ref, computed } from 'vue'
import { invoke } from '@tauri-apps/api/core'
import { listen } from '@tauri-apps/api/event'
import type { TickResult, ServiceStatus, AppError } from '@/types'

export const useServiceStore = defineStore('service', () => {
  // ── State ──────────────────────────────────────────────
  const running    = ref(false)
  const lastResult = ref<TickResult | null>(null)
  const logs       = ref<LogEntry[]>([])
  const loading    = ref(false)
  const error      = ref<string | null>(null)

  // ── Getters ────────────────────────────────────────────
  const statusText = computed(() => running.value ? '上报中' : '已停止')
  const lastAppName = computed(() => lastResult.value?.appName ?? '—')

  // ── Actions ────────────────────────────────────────────
  async function start() {
    loading.value = true
    error.value = null
    try {
      await invoke<ServiceStatus>('service_start')
      running.value = true
    } catch (e) {
      error.value = (e as AppError).message
    } finally {
      loading.value = false
    }
  }

  async function stop() {
    await invoke('service_stop')
    running.value = false
  }

  // ── 事件监听（应在 App.vue 初始化时调用一次）──────────────
  async function setupListeners() {
    await listen<TickResult>('service:tick', (event) => {
      lastResult.value = event.payload
    })
    await listen<LogEntry>('service:log', (event) => {
      logs.value.unshift(event.payload)
      if (logs.value.length > 500) logs.value.length = 500
    })
    await listen<{ running: boolean }>('service:status', (event) => {
      running.value = event.payload.running
    })
  }

  return { running, lastResult, logs, loading, error, statusText, lastAppName, start, stop, setupListeners }
})
```

### 4.2 命名约定

| 层 | 约定 |
|----|------|
| Store 文件名 | 小写单词，如 `service.ts` `config.ts` |
| Store ID | 与文件名同名字符串 |
| State | `ref()` / `reactive()`，**不用选项式 state** |
| Getter | `computed()`，命名用名词或形容词 |
| Action | 动词开头，如 `start()` `fetchMetrics()` `setTheme()` |

## 五、Composable 规范

### 5.1 命名约定

- 文件名：`use{功能名}.ts`，大驼峰
- 返回值：纯响应式数据 + 操作函数，**不返回 Store 实例**

```typescript
// src/composables/useConfig.ts

import { ref, readonly } from 'vue'
import { invoke } from '@tauri-apps/api/core'
import type { AppConfig } from '@/types/config'

export function useConfig() {
  const config = ref<AppConfig | null>(null)
  const loading = ref(false)

  async function loadAll() {
    loading.value = true
    config.value = await invoke<AppConfig>('config_get_all')
    loading.value = false
  }

  async function set<K extends keyof AppConfig>(key: K, value: AppConfig[K]) {
    await invoke('config_set', { key, value })
    if (config.value) config.value[key] = value
  }

  return {
    config: readonly(config),
    loading: readonly(loading),
    loadAll,
    set,
  }
}
```

### 5.2 生命周期管理

事件监听类 Composable 必须在 `onUnmounted` 中清理监听器：

```typescript
import { onMounted, onUnmounted } from 'vue'
import { listen } from '@tauri-apps/api/event'

export function useMetrics() {
  const metrics = ref<SystemMetrics | null>(null)
  let unlisten: (() => void) | null = null

  onMounted(async () => {
    unlisten = await listen<SystemMetrics>('metrics:update', (e) => {
      metrics.value = e.payload
    })
  })

  onUnmounted(() => {
    unlisten?.()
  })

  return { metrics }
}
```

## 七、`invoke()` 封装规范

不允许在组件或 View 中直接裸写 `invoke()`，必须封装到 Composable 或 Store：

```typescript
// ❌ 错误：在 View 中直接调用
const start = () => invoke('service_start')

// ✅ 正确：通过 Composable 或 Store
const store = useServiceStore()
const start = () => store.start()
```

### 7.1 命令封装层（可选，中等规模项目推荐）

```typescript
// src/api/commands.ts
// 统一封装所有 invoke 调用，提供类型安全

import { invoke } from '@tauri-apps/api/core'
import type { AppConfig, TickResult, SystemMetrics, UpdateInfo } from '@/types'

export const commands = {
  // 配置
  configGetAll:  ()                => invoke<AppConfig>('config_get_all'),
  configSet:     (key: string, value: unknown) => invoke('config_set', { key, value }),
  configSetMany: (obj: Partial<AppConfig>)     => invoke('config_set_many', { obj }),

  // 服务
  serviceStart:      () => invoke('service_start'),
  serviceStop:       () => invoke('service_stop'),
  serviceStatus:     () => invoke<{ running: boolean }>('service_status'),
  serviceLastResult: () => invoke<TickResult | null>('service_last_result'),

  // 系统
  systemGetMetrics:      () => invoke<SystemMetrics>('system_get_metrics'),
  systemCheckPermissions: () => invoke<Record<string, string>>('system_check_permissions'),
  systemGetFonts:         () => invoke<string[]>('system_get_fonts'),

  // 自启
  autostartEnable:    () => invoke('autostart_enable'),
  autostartDisable:   () => invoke('autostart_disable'),
  autostartIsEnabled: () => invoke<boolean>('autostart_is_enabled'),

  // 更新
  updateCheck:    (channel: string) => invoke<UpdateInfo | null>('update_check', { channel }),
  updateDownload: ()                => invoke('update_download'),
  updateInstall:  ()                => invoke('update_install'),

  // 隐私
  privacyPickWindow: () => invoke<WindowInfo | null>('privacy_pick_window'),
}
```

## 八、组件规范

### 8.1 SFC 结构顺序

```vue
<script setup lang="ts">
// 1. 导入（类型、Composable、Store、组件）
// 2. Props / Emits 定义
// 3. Store / Composable 使用
// 4. 响应式数据（ref / computed）
// 5. 函数定义
// 6. 生命周期钩子
</script>

<template>
  <!-- 单一根元素，语义化 HTML -->
</template>

<style scoped>
/* 只写该组件专属样式 */
/* 颜色/间距全用 var(--token-name) */
</style>
```

### 8.2 Props 规范

```typescript
// ✅ 正确：明确类型 + 默认值
interface Props {
  label:    string
  variant?: 'primary' | 'secondary' | 'danger'
  disabled?: boolean
}
const props = withDefaults(defineProps<Props>(), {
  variant:  'primary',
  disabled: false,
})

// ❌ 错误：不定义类型
defineProps(['label', 'variant'])
```

### 8.3 禁止事项

```vue
<!-- ❌ 禁止内联样式 -->
<div :style="{ color: seedColor, fontSize: '14px' }">...</div>

<!-- ✅ 正确：用 CSS 变量或 class -->
<div class="status-label">...</div>

<!-- ❌ 禁止在模板中写业务逻辑 -->
<button @click="invoke('service_start').then(() => running = true)">...</button>

<!-- ✅ 正确：事件处理函数写在 script -->
<button @click="serviceStore.start()">...</button>
```

## 九、初始化流程（App.vue）

```vue
<!-- src/App.vue -->
<script setup lang="ts">
import { onMounted } from 'vue'
import { useConfigStore } from '@/stores/config'
import { useServiceStore } from '@/stores/service'
import { useMetricsStore } from '@/stores/metrics'
import { useTheme } from '@/composables/useTheme'

const configStore  = useConfigStore()
const serviceStore = useServiceStore()
const metricsStore = useMetricsStore()
const { applyTheme } = useTheme()

onMounted(async () => {
  // 1. 加载配置（必须第一步）
  await configStore.loadAll()

  // 2. 应用主题（依赖配置）
  applyTheme(configStore.config)

  // 3. 注册后端事件监听
  await serviceStore.setupListeners()
  await metricsStore.setupListeners()

  // 4. 如果配置了启动时自动开启上报
  if (configStore.config.enableAutoServiceStart) {
    await serviceStore.start()
  }
})
</script>

<template>
  <AppShell>
    <RouterView />
  </AppShell>
</template>
```

### 10.2 Composable 测试示例

```typescript
// src/composables/__tests__/useTheme.test.ts

import { describe, it, expect, beforeEach } from 'vitest'
import { setActivePinia, createPinia } from 'pinia'
import { useTheme } from '@/composables/useTheme'

describe('useTheme', () => {
  beforeEach(() => {
    setActivePinia(createPinia())
    document.documentElement.removeAttribute('data-theme')
  })

  it('applyTheme dark 应设置 data-theme=dark', () => {
    const { applyTheme } = useTheme()
    applyTheme({ themeMode: 'dark', seedColor: '#06b6d4', uiScale: 100, uiFont: '' } as any)
    expect(document.documentElement.getAttribute('data-theme')).toBe('dark')
  })

  it('seedColor 应写入 CSS 变量', () => {
    const { applyTheme } = useTheme()
    applyTheme({ themeMode: 'light', seedColor: '#f59e0b', uiScale: 100, uiFont: '' } as any)
    const val = getComputedStyle(document.documentElement).getPropertyValue('--color-primary').trim()
    expect(val).toBe('#f59e0b')
  })
})
```
